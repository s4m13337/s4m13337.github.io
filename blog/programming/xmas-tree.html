Here is the final code that generates the above animation:

java Animate[ Overlay[{ Rotate[ Dynamic@ListPlot[ Table[{RandomReal[]*10, RandomReal[]*10}, {i, 1, 20},PlotMarkers -> Style["*", 18], Background -> Black, Axes -> False, PlotStyle -> White ], -90 \[Degree] ], Rotate[ Show[ Plot[x Sin[x + a], {x, 0, 30}, PlotStyle -> {Red, Thick, Dashed} ], Plot[-x Sin[x + a], {x, 0, 30}, PlotStyle -> {Darker@Green, Thick, Dashed} ], Axes -> False, Background -> Transparent ], -90 \[Degree] ] }], {a, 1, 0.01} ]
